    <script>
        // 眼睛跟随鼠标移动
        document.addEventListener('mousemove', (e) => {
            const eyes = document.querySelectorAll('.eye::after');
            const rect = document.querySelector('.left-panel').getBoundingClientRect();
            const x = (e.clientX - rect.left) / rect.width;
            const y = (e.clientY - rect.top) / rect.height;

            document.querySelectorAll('.eye').forEach(eye => {
                const pupil = eye.querySelector('::after') || eye;
                pupil.style.transform = `translate(${x * 10}px, ${y * 10}px)`;
            });
        });

        // 密码框聚焦时，黄色小人转头回避
        const passwordInput = document.getElementById('password');
        const yellowChar = document.querySelector('.char-yellow');

        passwordInput.addEventListener('focus', () => {
            yellowChar.style.transform = 'rotateY(180deg)';
        });

        passwordInput.addEventListener('blur', () => {
            yellowChar.style.transform = 'rotateY(0deg)';
        });

        // 密码显示/隐藏切换
        const togglePassword = document.getElementById('togglePassword');
        togglePassword.addEventListener('click', () => {
            const type = passwordInput.type === 'password' ? 'text' : 'password';
            passwordInput.type = type;
            togglePassword.textContent = type === 'password' ? '👁️' : '🙈';
        });
    </script>
